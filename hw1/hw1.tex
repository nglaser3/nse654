\section{Question 1}
To begin, the angular flux is:
\begin{align*}
  \psi(x, \mu) &= \sin(\frac{x}{4})\left[1 + \left(\mu-\frac{1}{10}x\right)^2\right]
\end{align*}

\subsection{Part A}
The scalar flux is the integral with respect to angle
\begin{align*}
  \phi(x) &= \int_{-1}^1 \dd{\mu}\sin(\frac{x}{4})\left[1 + \left(\mu-\frac{1}{10}x\right)^2\right]\\
  &= \sin(\frac{x}{4}) \int_{-1}^1 \dd{\mu}\left[1 + \left(\mu-\frac{1}{10}x\right)^2\right]\\ 
  &= \sin(\frac{x}{4})\left(\frac{8}{3} + \frac{x^2}{50}\right)

\section{Question 2}
To begin, the transport equation for this question can be simplified to:
\begin{align*}
  \mu \pdv{\psi_1}{x} &= -\sigma_{t, 1}\psi_1\\
  \mu \pdv{\psi_2}{x} &= -\sigma_{t, 2}\psi_2
\end{align*}

with the following boundary conditions
\begin{align*}
  \psi_1(0) &= \psi_0\\
  \psi_1(a) &= \psi_2(a)\\
  j_1(a) &= j_2(b)\\
  \psi_2(\infty) &= 0
\end{align*}

So then the general solutions are 
\begin{align*}
  \psi_1(x) &= \psi_0e^{-\sigma_{t,1} x}\\
  \psi_2(x) &= \psi_0e^{-\sigma_{t,1} a}e^{-\sigma_{t,2} (x-a)}
\end{align*}

Then the mean free path is the first moment of the pdf:
\begin{align*}
  mfp &= \int_0^{\infty}\dd{x} x\left[\sigma_{t,1}\psi_1(x) + \sigma_{t,2}\psi_2(x)\right]\\
  &= \int_0^{a}\dd{x} x\sigma_{t,1}\psi_1(x) + \int_a^{\infty}\dd{x} x \sigma_{t,2}\psi_2(x)\\
  &= \int_0^{a}\dd{x} x\sigma_{t,1}e^{-\sigma_{t,1} x} + \int_a^{\infty}\dd{x} x \sigma_{t,2} e^{-\sigma_{t,1} a}e^{-\sigma_{t,2}(x-a)}\\
  &= \frac{e^{-a\sigma_{t,1}}\left(\frac{\sigma_{t,1}}{\sigma_{t,2}}-1\right)+1}{\sigma_{t,1}}
\end{align*}

Then, plugging in an infinite optical thickness for region 1, we obtain
\begin{align*}
  mfp &= \frac{e^{-\infty}\left(\frac{\sigma_{t,1}}{\sigma_{t,2}}-1\right)+1}{\sigma_{t,1}}\\
  &= \frac{1}{\sigma_{t,1}}
\end{align*}

Then, plugging in $\sigma_{t,2}=\sigma_{t,1}$:
\begin{align*}
  &= \frac{e^{-a\sigma_{t,1}}\left(1-1\right)+1}{\sigma_{t,1}}\\
  &= \frac{1}{\sigma_{t,1}}
\end{align*}

\section{Question 3}
\subsection{Part A}
The angular flux at the spaceship should just be $\psi(E)$ when $\hat{\Omega}\cdot\hat{e}_s$ is greater than 0. This occurs when $\theta^*$ is less than $\arcsin(\frac{R_0}{\abs{s}})$, or equivalently, when $\mu^*$ is greater than $\cos(\arcsin(\frac{R_0}{\abs{s}}))=\sqrt{1-\sin^2(\arcsin(\frac{R_0}{\abs{s}}))}=\sqrt{1-\frac{R_0^2}{\abs{s}^2}}$
\begin{align}
  \psi(\vec{r},\hat{\Omega}, E) &= 
    \begin{cases}
        \psi(E),\ \abs{\mu} \geq \sqrt{1-\frac{R_0^2}{\abs{s}^2}}
    \end{cases}
\end{align}

\subsection{Part B}
The scalar flux is the integral of the angular flux over the solid angle, bounded by the previous restrictions on $\mu$

\begin{align*}
  \phi(\vec{r}, E) &= 2\int_{\sqrt{1-\frac{R_0^2}{\abs{s}^2}}}^1\dd{\mu}\psi(E)\\
  &= 2\psi(E)\left(1 - \sqrt{1-\frac{R_0^2}{\abs{s}^2}}\right) 
\end{align*}

I was unsure how to extend this into the azimuthal axis, without making the patch a square. I was thinking to do bounds for the azimuthal integral as dependent on $theta$, such that the min and max were $\theta^* - \theta$, which I think would trace the outer edge of the circlular patch, but again was unsure. 

\subsection{Part C}
The angular flux with the circumstellar medium is the same as without, just attenuated the distance $d$ through the medium:

\begin{align*}
  \psi(\vec{r},\hat{\Omega}, E) &= e^{-d\sigma_t(E)}\psi(E)
\end{align*}

To find $d$, we need to use the law of cosines, to find the total length of the extra side (from where the right angle is formed in the diagram to the space ship) and the vacuum component (where $R_1$ intersects the outer surface in the diagram). For the total length, the formula is, also using the quadratic formula,
\begin{align*}
  0 &= d_t^2 -2d_t\abs{\vec{s}}\cos(\theta^*) + \abs{\vec{s}}^2 - R_0^2\\
  d_t &= \frac{2\abs{\vec{s}}\cos(\theta^*) + \sqrt{4\abs{\vec{s}}^2\cos^2(\theta^*) - 4(\abs{\vec{s}}^2 - R_0^2)}}{2}
\end{align*}
And then the vacuum component
\begin{align*}
  0 &= d_v^2 -2d_v\abs{\vec{s}}\cos(\theta^*) + \abs{\vec{s}}^2 - R_1^2\\
  d_v &= \frac{2\abs{\vec{s}}\cos(\theta^*) + \sqrt{4\abs{\vec{s}}^2\cos^2(\theta^*) - 4(\abs{\vec{s}}^2 - R_1^2)}}{2}
\end{align*}

Then, subtracting the vacuum from the total, we find $d$ to be
\begin{align*}
  d &= \frac{2\abs{\vec{s}}\cos(\theta^*) + \sqrt{4\abs{\vec{s}}^2\cos^2(\theta^*) - 4(\abs{\vec{s}}^2 - R_0^2)}}{2} - \frac{2\abs{\vec{s}}\cos(\theta^*) + \sqrt{4\abs{\vec{s}}^2\cos^2(\theta^*) - 4(\abs{\vec{s}}^2 - R_1^2)}}{2} \\
  &= \frac{\sqrt{4\abs{\vec{s}}^2\cos^2(\theta^*) - 4(\abs{\vec{s}}^2 - R_0^2)} - \sqrt{4\abs{\vec{s}}^2\cos^2(\theta^*) - 4(\abs{\vec{s}}^2 - R_1^2)}}{2}
\end{align*}

subject to the same constrains on $\mu$ as part A.
